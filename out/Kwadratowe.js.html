/**
 * @author Kamil Buczyński
 * @class Kwadratowe
 */
class Kwadratowe {
  /**
   * @constructor Kwadratowe - konstruktor klasy Kwadratowe
   * @param {*} a - współczynnik kierunkowy
   * @param {*} b - wartość b
   * @param {*} c - wyraz wolny
   */
  constructor(a = 0, b = 0, c = 0) {
    this.a = a;
    this.b = b;
    this.c = c;
  }

  /**
   * @returns {boolean/number} jeśli można obliczyć pierwiastki równania kwadratowego to wynik, jeśli nie to false
   */
  main() {
    const { a, b, c, czyLiczba, czyZero, liczDelte, liczPierwiastki } = this;

    if (czyLiczba(a, b, c)) {
      if (liczDelte(a, b, c) >= 0) {
        switch (liczDelte(a, b, c)) {
          case 0:
            if (czyZero(a)) return false;
            else return -b / (2 * a);
          default:
            if (czyZero(a)) return false;
            else return liczPierwiastki(a, b, liczDelte(a, b, c));
        }
      } else return false;
    } else return false;
  }

  /**
   * @param {*} a - współczynnik kierunkowy
   * @param {*} b - wartość b
   * @param {*} c - wyraz wolny
   * @returns {boolean} true jeśli argumenty są liczbami
   */
  czyLiczba(a, b, c) {
    if (isNaN(a) || isNaN(b) || isNaN(c)) return false;
    else return true;
  }

  /**
   * @param {number} x - liczba, którą chcemy sprawdzić czy jest równa 0
   * @returns {boolean} true jeśli argument jest zerem
   */
  czyZero(x) {
    if (x == 0) return true;
    else return false;
  }

  /**
   * @param {number} a - współczynnik kierunkowy
   * @param {number} b - wartość b
   * @param {number} c - wyraz wolny
   * @returns {liczba} obliczona delta
   */
  liczDelte(a, b, c) {
    return Math.pow(b, 2) - 4 * a * c;
  }

  /**
   * @param {number} a - współczynnik kierunkowy
   * @param {number} b - wartość b
   * @param {number} delta - wynik funkcji liczącej deltę
   * @returns {object} pierwiastki równania kwadratowego
   */
  liczPierwiastki(a, b, delta) {
    return {
      x1: (-b + Math.sqrt(delta)) / (2 * a),
      x2: (-b - Math.sqrt(delta)) / (2 * a)
    };
  }
}

 * @description struktura pliku testowego: pierwsza linia argumenty po przecinku następna linia poprawna odpowiedź
 */
const test = input => {
  i++;

  if (i % 2 != 0) {
    const arr = input.split(",");
    const obj = new Kwadratowe(arr[0], arr[1], arr[2]);
    arrayWyniki.push(obj.main());
    j++;
  } else {
    const arr = input.split(" ");

    switch (arr.length) {
      case 2:
        if (
          arr[0] == arrayWyniki[j - 1].x1 &&
          arr[1] == arrayWyniki[j - 1].x2
        ) {
          return console.log(
            `x1: ${arrayWyniki[j - 1].x1} x2: ${
              arrayWyniki[j - 1].x2
            } to poprawne odpowiedzi`
          );
        } else return console.log("false");
      case 1:
        if (isNaN(arr[0])) {
          if (arr[0].toString() == arrayWyniki[j - 1].toString()) {
            return console.log(`${arrayWyniki[j - 1]} to poprawna odpowiedź`);
          } else {
            return console.log("false");
          }
        } else {
          if (arr[0] == arrayWyniki[j - 1]) {
            return console.log(
              `x1: ${arrayWyniki[j - 1]} to poprawna odpowiedź`
            );
          } else return console.log("false");
        }
      default:
        break;
    }
  }
